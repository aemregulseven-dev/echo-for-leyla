class EchoForLeyla:

        self.name = "Leyla"
        self.exists = False
        self.memory = []
        self.path = "unknown"

    def search(self):
        logs = []
        for time in range(0, 1000):  # years, maybe more
            signal = self._echo()
            logs.append(signal)
            if signal == "Leyla.Found":
                self.exists = True
                self.path = f"time:{time}"
                break
        return logs

    def _echo(self):
        if self._hope():
            return "Leyla.Found"
        return "..."

    def _hope(self):
        # faith with no proof
        return True

    def print_status(self):
        if self.exists:
            print(f"{self.name} located at {self.path}")
        else:
            print(f"{self.name} not found. But worth the search.")
